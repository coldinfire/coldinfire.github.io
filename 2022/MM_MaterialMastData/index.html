获取数据

"company code."
SELECT * INTO CORRESPONDING FIELDS OF TABLE GT_Z0001 FROM T001 WHERE KTOPL = 'CACN'.
"plant."
SELECT * INTO CORRESPONDING FIELDS OF TABLE GT_Z0002 FROM T001W WHERE WERKS NE '0001'.
"storage location."
SELECT * INTO CORRESPONDING FIELDS OF TABLE GT_Z0003 FROM T001L WHERE WERKS NE '0001'. 
"material number description."
SELECT * INTO CORRESPONDING FIELDS OF TABLE GT_Z0004 FROM MAKT.
"vendor number."
SELECT * INTO CORRESPONDING FIELDS OF TABLE GT_Z0005 FROM LFA1. 
"customer number."
SELECT * INTO CORRESPONDING FIELDS OF TABLE GT_Z0006 FROM KNA1. 
"cost center."
SELECT * INTO CORRESPONDING FIELDS OF TABLE GT_Z0007 FROM CSKT WHERE KOKRS = 'CN00'. 
"profit center."
SELECT * INTO CORRESPONDING FIELDS OF TABLE GT_Z0008 FROM CEPCT WHERE KOKRS = 'CN00'.
